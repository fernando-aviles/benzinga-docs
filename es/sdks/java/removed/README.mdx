<div id="get">
  ## get
</div>

Obtiene los elementos eliminados del punto final v2

<div id="example-usage">
  ### Ejemplo de uso
</div>

```java
package hello.world;

import java.lang.Exception;
import org.benzinga.BZClient.Bzclient;
import org.benzinga.BZClient.models.operations.GetRemovedResponse;
import org.benzinga.BZClient.models.operations.Type;

public class Application {

    public static void main(String[] args) throws Exception {

        Bzclient sdk = Bzclient.builder()
                .apiKeyAuth("<YOUR_API_KEY_HERE>")
            .build();

        GetRemovedResponse res = sdk.removed().get()
                .page(700347L)
                .pageSize(558834L)
                .type(Type.OFFERINGS)
                .updated(521235L)
                .call();

        if (res.modelsRmvdJSON().isPresent()) {
            // manejar respuesta
        }
    }
}
```
